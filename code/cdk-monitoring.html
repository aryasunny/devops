CDK: Monitoring Stack

Alarms, dashboards, and alerting wired as code.

// monitoring-stack.ts
import { Stack, Duration } from "aws-cdk-lib";
import { Alarm, Metric, ComparisonOperator, Dashboard } from "aws-cdk-lib/aws-cloudwatch";
import { Topic } from "aws-cdk-lib/aws-sns";
import { SnsAction } from "aws-cdk-lib/aws-cloudwatch-actions";

export class MonitoringStack extends Stack {
  constructor(scope: Construct, id: string, props: StackProps) {
    super(scope, id, props);

    const alerts = new Topic(this, "AlertsTopic");

    const lambdaErrors = new Metric({
      namespace: "AWS/Lambda",
      metricName: "Errors",
      statistic: "Sum",
      period: Duration.minutes(5),
    });

    const errAlarm = new Alarm(this, "LambdaErrorAlarm", {
      metric: lambdaErrors,
      threshold: 1,
      evaluationPeriods: 1,
      comparisonOperator: ComparisonOperator.GREATER_THAN_OR_EQUAL_TO_THRESHOLD,
      alarmDescription: "Lambda errors > 0 in 5 minutes",
    });
    errAlarm.addAlarmAction(new SnsAction(alerts));

    new Dashboard(this, "OpsDashboard", { dashboardName: "EmailService-Monitoring" });
  }
}
